/*
**
 * ClipSend - Sends text via clipboard, optionally restoring the original clipboard content.
 * @param toSend The text to send.
 * @param endChar Optional character to append (default: "").
 * @param isClipReverted If true, restores the original clipboard after sending (default: true).
 * @param untilRevert Delay in ms before restoring the clipboard (default: 100).
 * @returns {Boolean} True if successful, false if ClipWait failed.
*/
ClipSend(toSend, endChar := "", isClipReverted := true, untilRevert := 100) {
    if isClipReverted
        prevClip := ClipboardAll()

    A_Clipboard := toSend . endChar

    if !ClipWait(1) {
        if isClipReverted
            A_Clipboard := prevClip
        return false
    }

    Send("^v")

    if isClipReverted
        SetTimer(() => A_Clipboard := prevClip, -untilRevert)

    return true
}

;============================================================================
; Dash Maker - Type "dash" + character + number to repeat that character
; Examples: dash-10 → ----------, dash=5 → =====, dash#20 → ####################
; Supported: - = # * ~ + x
; End input with: Space, Enter, or Tab
;============================================================================

global MAX_REPEAT := 200  ; Maximum repetitions allowed

; Register hotstrings for each character
for item in ['-', '=', '#', '*', '~', '+', 'x']
    HotString(':*:dash' item, StartDashCapture.Bind())

StartDashCapture(*) {
    global thisString := ""
    global dashInputHook := InputHook('I2')

    dashInputHook.OnChar := OnDashChar
    dashInputHook.OnKeyUp := OnDashEnd
    dashInputHook.KeyOpt('{Space}{Enter}{Tab}', '+N')  ; End keys
    dashInputHook.Start()
}

OnDashChar(ih, char) {
    global thisString .= char
}

OnDashEnd(ih, vk, sc) {
    global thisString, MAX_REPEAT

    ; Get the character from the hotstring (e.g., "dash-" → "-")
    dashChar := '{' SubStr(A_ThisHotkey, -1) '}'

    ; Build output based on input
    if (IsNumber(thisString) && thisString > 0 && thisString <= MAX_REPEAT) {
        output := ""
        Loop thisString
            output .= dashChar
    } else {
        ; Not a valid number, output original text
        output := "dash" dashChar thisString
    }

    SendInput(output)
    ih.Stop()
    thisString := ""
}

^!d::

    yDay := A_Now

    ; Calculate the previous working date
    if (A_WDay = 1)
        yDay := DateAdd(yDay, -2, "days")
    else if (A_WDay = 2)
        yDay := DateAdd(yDay, -3, "days")
    else
        yDay := DateAdd(yDay, -1, "days")

    yDay := FormatTime(yDay, "dd.MM.yyyy")
    tDay := FormatTime(A_Now, "dd.MM.yyyy")

    ; Delete the existing date range and fill new one
    Send "{BS 24}" yDay " to " tDay
    return
}
